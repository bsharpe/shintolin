def html_messages(user_id, x, y, z)
  user = User.new(user_id)
  messages = mysql_get_messages(x, y, z, user)
  html = ''
  messages.each do
    |row| 
      html += "\n\t<div class='#{row['type']}'>" +
      "#{describe_message(row, user_id)}</div>"
  end
  html
end
