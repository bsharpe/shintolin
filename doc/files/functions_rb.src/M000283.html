def tick_grow_fields
  return "Crops only grow in Summer" if season != :Summer
  tiles = mysql_select('grid',{'terrain'=>91})
  tiles.each do
    |tile|
    growth = tile['hp'].to_i * 3
    mysql_bounded_update('grid', 'building_hp', 
      {'x'=>tile['x'], 'y'=>tile['y']}, +growth, 200)
  end  
end
