  def supporters
    result = mysql_select('accounts',
      {'settlement_id'=>settlement_id,'vote'=>mysql_id})
    supporters = []
    result.each {|row| supporters << User.new(row['id'])}
    supporters.delete_if {|user| user.hp == 0 || user.active == 0}
    supporters.count
  end
